fname = 'Hello.txt'

### Pause time in seconds:
hold = 1

### Voice Options:
	# Female---
	# Kathy (hilarious sounding),
	# Vicki (seductive/Russian af),
	# Victoria (kinda normal),

	# Male---
	# Alex (Okay)
	# Bruce (Okay)
	# Fred (Pretty damn funny)
voice = 'Vicki'

### You can ignore this:
from os import system
import time

voice = voice.lower().capitalize()
if voice not in ['Kathy', 'Vicki','Victoria',
'Alex', 'Bruce',' Fred']:
	voice = 'Vicki'

with open(fname) as f:
	lines = [line.strip() for line in f.readlines()]

for l in lines:
	if l != '%':
		print(l)
		system ('say -v %s %s' %(voice, l))
	else:
		print ""
		time.sleep(hold)
