function [X,Y] = euler(x,y,x1,n)

h = (x1-x)/n;
%step size

x = x;
y = y;
%initial conditions

for i = 1:n
%Euler iteration

    y = y + h*f(x,y);
    x = x + h;
    %new x and y
    
    X = [X;x];
    Y = [Y;y];
    %updates new values
    
end

end
